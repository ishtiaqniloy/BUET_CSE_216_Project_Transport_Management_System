UPDATE LICENSESERVICEHISTORY LH
SET LH.SERVICE_DATE = (SELECT L.ISSUE_DATE FROM LICENSE L WHERE L.LICENSE_NO =  LH.LICENSE_NO ) + 3
WHERE (SELECT L.ISSUE_DATE FROM LICENSE L WHERE L.LICENSE_NO =  LH.LICENSE_NO) > LH.SERVICE_DATE;

UPDATE PAYMENT P
SET PAYMENT_DATE = ( SELECT L.SERVICE_DATE FROM LICENSESERVICEHISTORY L WHERE P.HISTORY_NO = L.ENTRY_NO) + 3
WHERE P.HISTORY_NO < 2000 AND P.PAYMENT_DATE < ( SELECT L.SERVICE_DATE FROM LICENSESERVICEHISTORY L WHERE P.HISTORY_NO = L.ENTRY_NO);

UPDATE VEHICLESERVICEHISTORY VH
SET VH.SERVICE_DATE = (SELECT V.REG_DATE FROM VEHICLE V WHERE V.REG_NO =  VH.VEHICLE_REG_NO ) + 3
WHERE (SELECT V.REG_DATE FROM VEHICLE V WHERE V.REG_NO =  VH.VEHICLE_REG_NO) > VH.SERVICE_DATE;

UPDATE PAYMENT P
SET PAYMENT_DATE = ( SELECT V.SERVICE_DATE FROM VEHICLESERVICEHISTORY V WHERE P.HISTORY_NO = V.ENTRY_NO) + 3
WHERE P.HISTORY_NO > 2000 AND P.HISTORY_NO <3000 AND P.PAYMENT_DATE < ( SELECT V.SERVICE_DATE FROM VEHICLESERVICEHISTORY V WHERE P.HISTORY_NO = V.ENTRY_NO);

UPDATE PENALTYHISTORY PH
SET PH.PENALTY_DATE = (SELECT L.ISSUE_DATE FROM LICENSE L WHERE L.LICENSE_NO =  PH.LICENSE_NO ) + 3
WHERE (SELECT L.ISSUE_DATE FROM LICENSE L WHERE L.LICENSE_NO =  PH.LICENSE_NO) > PH.PENALTY_DATE;

UPDATE PENALTYHISTORY PH
SET PH.PENALTY_DATE = (SELECT V.REG_DATE FROM VEHICLE V WHERE V.REG_NO =  PH.V_REG_NO ) + 3
WHERE (SELECT V.REG_DATE FROM VEHICLE V WHERE V.REG_NO =  PH.V_REG_NO) > PH.PENALTY_DATE;

UPDATE PAYMENT P
SET PAYMENT_DATE = ( SELECT PH.PENALTY_DATE FROM PENALTYHISTORY PH WHERE P.HISTORY_NO = PH.ENTRY_NO) + 3
WHERE P.HISTORY_NO > 3000 AND P.PAYMENT_DATE < ( SELECT PH.PENALTY_DATE FROM PENALTYHISTORY PH WHERE P.HISTORY_NO = PH.ENTRY_NO);
