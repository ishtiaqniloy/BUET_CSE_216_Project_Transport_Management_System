UPDATE LICENSE L
SET ISSUE_DATE = (SELECT BIRTH_DATE FROM DRIVER D WHERE D.LICENSE_NO = L.LICENSE_NO) + 18*365+1
WHERE L.ISSUE_DATE - (SELECT BIRTH_DATE FROM DRIVER D WHERE D.LICENSE_NO = L.LICENSE_NO) < 18*365;

UPDATE DRIVER D
SET BIRTH_DATE = BIRTH_DATE-2*365
WHERE (SELECT D.DRIVER_NID FROM LICENSE L WHERE D.LICENSE_NO = L.LICENSE_NO AND ISSUE_DATE > '30-Nov-2017') is not null;

Update LICENSE L
SET ISSUE_DATE = ISSUE_DATE - 2*365+2
WHERE ISSUE_DATE > '30-Nov-2017';

commit;
