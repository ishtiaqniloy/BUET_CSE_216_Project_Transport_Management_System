DELETE VEHICLESERVICEHISTORY V
WHERE (SELECT TYPE FROM SERVICE S WHERE S.SID = V.SERVICE_ID) = 'L';

DELETE LICENSESERVICEHISTORY L
WHERE (SELECT TYPE FROM SERVICE S WHERE S.SID = L.SERVICE_ID) = 'V';

DELETE PAYMENT P
WHERE P.HISTORY_NO > 2000 AND P.HISTORY_NO < 3000 AND (SELECT ENTRY_NO FROM VEHICLESERVICEHISTORY V WHERE V.ENTRY_NO = P.HISTORY_NO) is NULL;

DELETE PAYMENT P
WHERE P.HISTORY_NO < 2000 AND (SELECT ENTRY_NO FROM LICENSESERVICEHISTORY L WHERE L.ENTRY_NO = P.HISTORY_NO) is NULL;
